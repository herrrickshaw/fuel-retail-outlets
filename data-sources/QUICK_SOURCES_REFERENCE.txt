💻 PYTHON SCRIPT TO AGGREGATE DATA
════════════════════════════════════════════════════════════════════════════

```python
import pandas as pd
import folium
from folium.plugins import MarkerCluster

# Load data from different sources
ppac_data = pd.read_excel('ppac_outlets.xlsx')
iocl_data = pd.read_csv('iocl_outlets.csv')
osm_data = pd.read_json('osm_outlets.json')

# Combine all datasets
all_outlets = pd.concat([ppac_data, iocl_data, osm_data])

# Remove duplicates (keeping official PPAC records)
unique_outlets = all_outlets.drop_duplicates(
    subset=['latitude', 'longitude', 'outlet_name'],
    keep='first'
)

print(f"Total unique outlets: {len(unique_outlets)}")

# Create map
m = folium.Map(
    location=[20.5, 78.5],
    zoom_start=5,
    tiles='CartoDB dark_matter'
)

# Add markers
marker_cluster = MarkerCluster().add_to(m)

for idx, row in unique_outlets.iterrows():
    folium.Marker(
        location=[row['latitude'], row['longitude']],
        popup=f"{row['outlet_name']}<br>{row['city']}",
        icon=folium.Icon(color='red')
    ).add_to(marker_cluster)

m.save('outlets_map.html')
print("Map saved to outlets_map.html")
```
